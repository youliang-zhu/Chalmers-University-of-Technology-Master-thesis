\chapter*{Nomenclature}
Below is the nomenclature of indices, sets, parameters, and variables used throughout this thesis.
% \vspace*{-3cm}

\section*{Indices}
\renewcommand*{\arraystretch}{1.35}
\begin{longtable}{p{0.2\textwidth}p{0.68\textwidth}}
$i$ & Index for a retrieved session or chunk\\
$r$ & Index for an iterative retrieval round\\
$t$ & Timestamp or temporal reference\\
\end{longtable}

% \vspace*{-3cm}
\section*{Sets}
\renewcommand*{\arraystretch}{1.35}
\begin{longtable}{p{0.2\textwidth}p{0.68\textwidth}}
$\mathcal{C}$ & Set of retrieval cues generated by the Planner\\
$\mathcal{E}$ & Set of synthesized evidence events\\
$\mathcal{S}$ & Set of dialogue sessions or retrieved memory chunks\\
$\mathcal{T}$ & Set of temporal expressions or normalized time spans\\
\end{longtable}

\section*{Parameters}
% \vspace*{-3cm}
\renewcommand*{\arraystretch}{1.35}
\begin{longtable}{p{0.2\textwidth}p{0.68\textwidth}}
$K$ & Number of retrieved sessions or chunks per retrieval cue\\
$R_{\max}$ & Maximum number of Planner-Synthesizer retrieval rounds\\
$\lambda$ & Weight assigned to an auxiliary reward component\\
\end{longtable}

\section*{Variables}
\renewcommand*{\arraystretch}{1.35}
\begin{longtable}{p{0.2\textwidth}p{0.68\textwidth}}
$Q$ & User query submitted to the memory system\\
$c_r$ & Retrieval cue generated at round $r$\\
$e_i$ & Synthesized evidence event\\
$s_i$ & Source session or memory chunk\\
$v$ & Synthesizer verdict, e.g. sufficient, indicative, or insufficient\\

\end{longtable}
